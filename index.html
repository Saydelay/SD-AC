async function pruefeCode() { const eingabe = document.getElementById("codeInput").value.trim().toLowerCase(); const ergebnis = document.getElementById("ergebnis"); if (!eingabe) { ergebnis.innerHTML = "❗ Bitte gib einen Code ein."; return; } try { const response = await fetch('codes.json'); const codes = await response.json(); if (codes.includes(eingabe)) { ergebnis.innerHTML = "⚠️ Verdächtiger Code gefunden!"; } else { ergebnis.innerHTML = "✅ Kein bekannter Cheat-Code."; } } catch { ergebnis.innerHTML = "❗ Fehler beim Laden der Codes."; } }
